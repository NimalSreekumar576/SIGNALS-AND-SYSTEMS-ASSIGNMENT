\textbf{Question 11.9.2.5:}
In an A.P., if the $p$-th term is $\frac{1}{q}$ and $q$-th term is $\frac{1}{p}$, prove that the sum of the first $pq$ terms is $\frac{1}{2}\brak{pq + 1}$, where $p \neq q$.\brak{\text{Using Z-transform of x\brak{n}}}.\\

\solution
\begin{align}
    \frac{1}{q} &= x\brak{0} + pd \\
    \frac{1}{p} &= x\brak{0} + qd
\end{align}

From equations \brak{1} and \brak{2}, the argumented matrix is:
\begin{align}
\begin{pmatrix}
1 & p & \frac{1}{q} \\
1 & q & \frac{1}{p}
\end{pmatrix}
\xleftrightarrow {R_2\leftarrow R_2-R_1}
\begin{pmatrix}
1 & p & \frac{1}{q} \\
0 & q-p & \frac{q-p}{pq}
\end{pmatrix}\\
\xleftrightarrow {R_2\leftarrow \frac{R_2}{q-p}}
\begin{pmatrix}
1 & p & \frac{1}{q} \\
0 & 1 & \frac{1}{pq}
\end{pmatrix}\\
\xleftrightarrow {R_1\leftarrow R_1-pR_2}
\begin{pmatrix}
1 & 0 & 0 \\
0 & 1 & \frac{1}{pq}
\end{pmatrix}\\
\implies 
\begin{pmatrix}
x\brak{0} \\
d
\end{pmatrix}
=
\begin{pmatrix}
0 \\
\frac{1}{pq}
\end{pmatrix}
\end{align}


\begin{align}
x\brak{n}&=\brak{\frac{n}{pq}}u \brak{n}\\
X\brak{z} &= \frac{z^{-1}}{pq\brak{1-z^{-1}}^2}  \qquad ,|z| > 1
\end{align}

Finding sum of $pq$ terms,
\begin{align}
y\brak{n} &=\frac{n}{2} \brak{2x\brak{0} + \brak{n+1}d}\\
y\brak{pq} &=\frac{1}{2}\brak{pq + 1}
\end{align}
\begin{tabular}{|c|c|c|}
    \hline
    \textbf{Symbols} & \textbf{Values} & \textbf{Description} \\
    \hline
    $x\brak{n}$ & $\brak{\frac{n}{pq}}u\brak{n}$ & general term of the series \\
    \hline
    $y\brak{n}$ & $ \frac{\brak{n+1}}{2}$ & sum of n terms \\
    \hline
\end{tabular}\\
